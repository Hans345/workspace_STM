22 // Initalisierung

28 float phi = (-2.0 * M_PI * k * n) / N;

47void fft(Complex c[], int N) {
48 if (N <= 1)
49 return;
50
51 // Aufteilen
52 Complex ce[N / 2];
53 Complex co[N / 2];
54 for (int i = 0; i < N / 2; i++) {
55 ce[i] = c[2 * i];
56 co[i] = c[2 * i + 1];
57 }
58
59 // Rekursion
60 fft(ce, N / 2);
61 fft(co, N / 2);
62
63 // Kombinieren
64 for (int k = 0; k < N / 2; k++) {
65 float angle = (-2.0 * M_PI * k) / N;
66 Complex twiddle = complex_exp(angle);
67
68 Complex t = complex_mul(twiddle, co[k]);
69
70 c[k] = complex_add(ce[k], t);
71 c[k + N / 2] = complex_sub(ce[k], t);
72 }
73}
